 #define MOTOR_PIN_DIR_REV     GPIO_PIN_1

 static uint8_t s_target_position = 0U;
 static uint8_t s_movement_active = 0U;
 void MotorController_Init(void)
 {
     /* Configure IN1 and IN2 as normal output */
     GPIO_InitTypeDef GPIO_InitStruct = {0};
     GPIO_InitStruct.Pin = MOTOR_PIN_DIR_FWD | MOTOR_PIN_DIR_REV;
     GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
     GPIO_InitStruct.Pull = GPIO_NOPULL;
     GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
     HAL_GPIO_Init(MOTOR_GPIO_PORT_DIR, &GPIO_InitStruct);

     /* SWE-REQ-009: Set outputs safe */
     s_motor_state = MOTOR_STATE_STOPPED;
     s_target_position = 0U;
     s_movement_active = 0U;

     /* Ensure PWM/GPIOs are safe */
     HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_FWD, GPIO_PIN_RESET);
     HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_REV, GPIO_PIN_RESET);

     /* Start PWM channel for motor */
     HAL_TIM_PWM_Start(&MOTOR_PWM_HANDLE, MOTOR_PWM_CHANNEL);
 }

 void MotorController_MoveTo(uint8_t target_position)
 {
     /* SWE-REQ-009: Initiate movement to target position */
     s_target_position = target_position;
     uint8_t current_position = 0;
     if(PositionSensing_GetPosition(&current_position))
     {
         if(current_position == s_target_position)
         {
             /* At target already, no move */
             MotorController_Abort();
             return;
         }
         /* Decide direction */
         if(current_position < s_target_position)
         {
             
             HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_FWD, GPIO_PIN_RESET);
             HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_REV, GPIO_PIN_SET);
             s_motor_state = MOTOR_STATE_MOVING_FWD;
         }
         else
         {
             HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_FWD, GPIO_PIN_SET);
             HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_REV, GPIO_PIN_RESET);
             s_motor_state = MOTOR_STATE_MOVING_REV;
         }
         s_movement_active = 1U;
         HAL_TIM_PWM_Start(&MOTOR_PWM_HANDLE, MOTOR_PWM_CHANNEL);
     }
     else
     {
         /* Invalid position, abort */
         MotorController_Abort();
     }
 }
 void MotorController_Update(void)
 {
     /* SWE-REQ-010: Stop at target or if invalid */
     if(s_movement_active)
     {
        PositionSensing_Update();
        uint8_t current_position = 0U;
        if(PositionSensing_GetPosition(&current_position))
         {
             
             if (PositionSensing_IsAtTarget(s_target_position))
		{
    			MotorController_Abort();
		}

         }
         else
         {
             /* Out-of-range, stop motor */
             MotorController_Abort();
         }
     }
 }
 void MotorController_Abort(void)
 {
     s_motor_state = MOTOR_STATE_STOPPED;
     s_movement_active = 0U;

     HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_FWD, GPIO_PIN_RESET);
     HAL_GPIO_WritePin(MOTOR_GPIO_PORT_DIR, MOTOR_PIN_DIR_REV, GPIO_PIN_RESET);
     HAL_TIM_PWM_Stop(&MOTOR_PWM_HANDLE, MOTOR_PWM_CHANNEL);
 }

         (s_target_position < 6U) &&

RELEVANT SECURE-C RULES:


Rule ID: CON33-C

Category: CON

Description:
Avoid race conditions when using library functions by ensuring that shared resources are accessed in a thread-safe manner. Library functions are not always reentrant or thread-safe, so protect calls to such functions with appropriate synchronization mechanisms to prevent concurrent access that can lead to inconsistent or corrupted data.

Fix:
Use mutexes or disable interrupts around library function calls that access shared resources to ensure exclusive access, especially in embedded or RTOS-based STM32 applications.



Rule ID: DCL40-C

Category: DCL

Description:
Do not create incompatible declarations of the same function or object. Incompatible declarations can lead to undefined behavior due to mismatched types, calling conventions, or storage durations. Ensure all declarations and definitions consistently use the same type, qualifier, and linkage.

Fix:
Ensure all declarations and definitions of a function or object use exactly the same type, qualifiers, and linkage specifications. In STM32 projects, declare functions in header files with matching prototypes and include these headers consistently across source files.



Rule ID: MSC40-C

Category: MSC

Description:
Do not violate constraints imposed by the language or the implementation, such as accessing memory out of bounds, using uninitialized variables, or exceeding hardware limitations. Such violations can lead to undefined behavior, system crashes, or security vulnerabilities.

Fix:
Validate all inputs, initialize variables before use, and enforce memory boundaries using static or dynamic checks appropriate for embedded environments.



Rule ID: MSC37-C

Category: MSC

Description:
Ensure that control never reaches the end of a non-void function without an explicit return statement, to prevent undefined behavior and ensure deterministic function outcomes.

Fix:
Add an explicit return statement with an appropriate value before the end of all non-void functions.



Rule ID: ERR33-C

Category: ERR

Description:
Detect and properly handle error indications returned by standard library functions to avoid undefined behavior or security vulnerabilities caused by ignoring errors.

Fix:
Check return values of standard library functions, handle errors immediately by implementing recovery, cleanup, or safe fallback procedures suitable for STM32 embedded environments.
